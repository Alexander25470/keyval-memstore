using System.Text;
using System.Linq;
using System.Text;
using KeyValueStore.Server;
using KeyValueStore.Server.Networking;
using KeyValueStore.Server.PubSub;
using KeyValueStore.Server.Resp;
using KeyValueStore.Server.Store;

namespace KeyValueStore.Tests.PubSub;

public class PubSubCommandsTests
{
    private readonly InMemoryStore _store = new();
    private readonly PubSubHub _hub = new();
    private readonly CommandDispatcher _dispatcher;

    public PubSubCommandsTests()
    {
        _dispatcher = new CommandDispatcher(_store, _hub);
    }

    private async Task<string> Exec(string commandLine)
    {
        using var ms = new MemoryStream();
        var writer = new RespWriter(ms);
        var args = commandLine.Split(' ').Select(s => new ReadOnlyMemory<byte>(Encoding.ASCII.GetBytes(s))).ToArray();
        await _dispatcher.ExecuteAsync(args, writer);
        ms.Position = 0;
        return Encoding.UTF8.GetString(ms.ToArray());
    }

    [Fact] public async Task Publish_ReturnsZero() => Assert.Equal(":0\r\n", await Exec("PUBLISH foo msg"));
    [Fact] public async Task Publish_WrongArity_ReturnsError() => Assert.StartsWith("-ERR", await Exec("PUBLISH foo"));
    [Fact] public async Task Subscribe_WrongArity_ReturnsError() => Assert.StartsWith("-ERR", await Exec("SUBSCRIBE"));
    [Fact] public async Task Subscribe_WithoutSession_ReturnsError() => Assert.StartsWith("-ERR", await Exec("SUBSCRIBE foo"));
    [Fact] public async Task PSubscribe_WrongArity_ReturnsError() => Assert.StartsWith("-ERR", await Exec("PSUBSCRIBE"));
    [Fact] public async Task PSubscribe_WithoutSession_ReturnsError() => Assert.StartsWith("-ERR", await Exec("PSUBSCRIBE foo"));
    [Fact] public async Task Unsubscribe_WithoutSession_ReturnsError() => Assert.StartsWith("-ERR", await Exec("UNSUBSCRIBE"));
    [Fact] public async Task PUnsubscribe_WithoutSession_ReturnsError() => Assert.StartsWith("-ERR", await Exec("PUNSUBSCRIBE"));

    // ---- successful subscription paths (via stub session) ----

    [Fact] public async Task Subscribe_WithSession_SendsAck()
    {
        var session = new StubSession();
        using var ms = new MemoryStream();
        var writer = new RespWriter(ms);
        await _dispatcher.ExecuteAsync("SUBSCRIBE foo".Split(' '), writer, session);
        ms.Position = 0;
        var resp = Encoding.UTF8.GetString(ms.ToArray());
        Assert.StartsWith("*3", resp);
        Assert.Contains("subscribe", resp);
        Assert.Contains("foo", resp);
    }

    [Fact]
    public async Task Publish_WithSubscriber_ReturnsCount()
    {
        var session = new StubSession();
        using var ms = new MemoryStream();
        var writer = new RespWriter(ms);
        await _dispatcher.ExecuteAsync("SUBSCRIBE foo".Split(' '), writer, session);
        Assert.Equal(":1\r\n", await Exec("PUBLISH foo hi"));
    }

    [Fact]
    public async Task SubscribeMultiple_ThenUnsubscribeAll()
    {
        var session = new StubSession();
        using var ms = new MemoryStream();
        var writer = new RespWriter(ms);
        await _dispatcher.ExecuteAsync("SUBSCRIBE a".Split(' '), writer, session);
        await _dispatcher.ExecuteAsync("SUBSCRIBE b".Split(' '), writer, session);
        Assert.Equal(":1\r\n", await Exec("PUBLISH a msg"));
        session.EnterSubscriptionMode(SubscriptionMode.None);
        using var ms2 = new MemoryStream();
        var writer2 = new RespWriter(ms2);
        var unsub = "UNSUBSCRIBE".Split(' ').Select(s => new ReadOnlyMemory<byte>(Encoding.ASCII.GetBytes(s))).ToArray();
        await _dispatcher.ExecuteAsync(unsub, writer2, session);
        Assert.Equal(":0\r\n", await Exec("PUBLISH a msg"));
    }

    [Fact]
    public async Task PSubscribe_ThenPublish_MatchesPattern()
    {
        var session = new StubSession();
        using var ms = new MemoryStream();
        var writer = new RespWriter(ms);
        await _dispatcher.ExecuteAsync("PSUBSCRIBE orders.*".Split(' '), writer, session);
        Assert.Equal(":0\r\n", await Exec("PUBLISH foo msg"));      // no match
        Assert.Equal(":1\r\n", await Exec("PUBLISH orders.new msg")); // matches
    }

    [Fact]
    public async Task Unsubscribe_WithSpecificChannel()
    {
        var session = new StubSession();
        using var ms = new MemoryStream();
        var writer = new RespWriter(ms);
        await _dispatcher.ExecuteAsync("SUBSCRIBE a".Split(' '), writer, session);
        await _dispatcher.ExecuteAsync("SUBSCRIBE b".Split(' '), writer, session);
        using var ms2 = new MemoryStream();
        var writer2 = new RespWriter(ms2);
        await _dispatcher.ExecuteAsync("UNSUBSCRIBE a".Split(' '), writer2, session);
        Assert.Equal(":0\r\n", await Exec("PUBLISH a msg"));   // unsubscribed
        Assert.Equal(":1\r\n", await Exec("PUBLISH b msg"));   // still subbed
    }

    [Fact]
    public async Task PUnsubscribe_WithSpecificPattern()
    {
        var session = new StubSession();
        using var ms = new MemoryStream();
        var writer = new RespWriter(ms);
        await _dispatcher.ExecuteAsync("PSUBSCRIBE foo.*".Split(' '), writer, session);
        using var ms2 = new MemoryStream();
        var writer2 = new RespWriter(ms2);
        await _dispatcher.ExecuteAsync("PUNSUBSCRIBE foo.*".Split(' '), writer2, session);
        Assert.Equal(":0\r\n", await Exec("PUBLISH foo.bar msg"));
    }
}

internal class StubSession : ClientSession
{
    // ClientSession constructor only assigns fields, doesn't use them.
    public StubSession() : base(null!, null!, null!) { }
}
